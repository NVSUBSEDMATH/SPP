bestfit_at <- list(aicc=Inf)
for(i in 1:25)
{
  fit_at <- auto.arima(at, xreg=fourier(at, K=i), seasonal=FALSE)
  if(fit_at$aicc < bestfit_at$aicc)
    bestfit_at <- fit_at
  else break;
}
